     1Ôćĺ"""Content context ÔÇö ContentPack aggregate root.
     2Ôćĺ
     3ÔćĺA :class:`ContentPack` is a curated bundle of content artifacts (Concept
     4Ôćĺrevisions, new QuestionTemplates, new TemplateVersions, new
     5ÔćĺLearningObjectives/Misconceptions) staged for atomic publication as a

     7Ôćĺauthor assembles a pack, submits it for multi-stage review, and once
     8Ôćĺevery stage has approved it the pack is published ÔÇö atomically binding
     9Ôćĺits artifacts to a new (or existing) ContentVersion and marking that

    13Ôćĺ

    16Ôćĺa separate out-of-band review process with no audit trail. ContentPack

    33Ôćĺ

    35Ôćĺ

   515Ôćĺ    def submit_for_review(self) -> None:
   516Ôćĺ        """Transition the pack from ``draft`` to ``in_review``.
   517Ôćĺ
   518Ôćĺ        Pre-state: ``draft``.
   519Ôćĺ        Post-state: ``in_review`` with no approved stages (a fresh
   520Ôćĺ        review cycle starts here).
   521Ôćĺ
   522Ôćĺ        Raises:
   523Ôćĺ            ContentPackAlreadySubmitted: If the pack is already in
   524Ôćĺ                ``in_review``.
   525Ôćĺ            ContentPackAlreadyPublished: If the pack is published.
   526Ôćĺ            ContentPackAlreadyRejected: If the pack is rejected.
   527Ôćĺ            InvalidStateTransition: If the pack is in any other state.
   528Ôćĺ        """
   529Ôćĺ        self._assert_not_terminal()
   530Ôćĺ        if self._status == ContentStatus.IN_REVIEW:
   531Ôćĺ            raise ContentPackAlreadySubmitted(self._id)
   532Ôćĺ        self._assert_status(ContentStatus.DRAFT, "submit_for_review")
   533Ôćĺ
   534Ôćĺ        self._status = ContentStatus.IN_REVIEW
   535Ôćĺ        # A fresh submission clears any stale approvals from a prior cycle.
   536Ôćĺ        self._approved_stages = set()
   537Ôćĺ        self._record_event(
   538Ôćĺ            ContentPackSubmittedForReview(
   539Ôćĺ                content_pack_id=self._id,
   540Ôćĺ                subject_id=self._subject_id,
   541Ôćĺ                author_id=self._author_id,
   542Ôćĺ            )
   543Ôćĺ        )

   545Ôćĺ    def approve(self, stage: ReviewStage, reviewer_id: UserId) -> None:
   546Ôćĺ        """Record a reviewer's approval at ``stage``.
   547Ôćĺ
   548Ôćĺ        The pack stays in ``in_review`` status ÔÇö multiple stages must
   549Ôćĺ        approve (see :attr:`required_stages`) before :meth:`publish` is
   550Ôćĺ        allowed.
   551Ôćĺ
   552Ôćĺ        Args:
   553Ôćĺ            stage: The review stage being approved. Must be in
   554Ôćĺ                :attr:`required_stages`.
   555Ôćĺ            reviewer_id: The user recording the approval. Recorded on
   556Ôćĺ                the event for audit.
   557Ôćĺ
   558Ôćĺ        Raises:
   559Ôćĺ            ContentPackAlreadyPublished: If the pack is published.
   560Ôćĺ            ContentPackAlreadyRejected: If the pack is rejected.
   561Ôćĺ            ContentPackNotInReview: If the pack is in ``draft``.
   562Ôćĺ            ContentPackStageAlreadyApproved: If ``stage`` has already
   563Ôćĺ                approved this submission cycle.
   564Ôćĺ            InvariantViolation: If ``stage`` is not in
   565Ôćĺ                :attr:`required_stages`.
   566Ôćĺ        """
   567Ôćĺ        self._assert_not_terminal()
   568Ôćĺ        if self._status != ContentStatus.IN_REVIEW:
   569Ôćĺ            raise ContentPackNotInReview(self._id, self._status.value)
   570Ôćĺ        if stage not in self._required_stages:
   571Ôćĺ            raise InvariantViolation(
   572Ôćĺ                "ContentPack",
   573Ôćĺ                f"stage {stage!r} is not in required_stages {sorted(s.value for s in self._required_stages)!r}",
   574Ôćĺ            )
   575Ôćĺ        if stage in self._approved_stages:
   576Ôćĺ            raise ContentPackStageAlreadyApproved(self._id, stage)
   577Ôćĺ
   578Ôćĺ        self._approved_stages.add(stage)
   579Ôćĺ        self._record_event(
   580Ôćĺ            ContentPackApproved(
   581Ôćĺ                content_pack_id=self._id,
   582Ôćĺ                subject_id=self._subject_id,
   583Ôćĺ                stage=stage,
   584Ôćĺ                reviewer_id=reviewer_id,
   585Ôćĺ            )
   586Ôćĺ        )

   683Ôćĺ    def publish(
   684Ôćĺ        self,
   685Ôćĺ        content_version_id: ContentVersionId,
   686Ôćĺ        now: datetime | None = None,
   687Ôćĺ    ) -> None:
   688Ôćĺ        """Transition the pack from ``in_review`` to ``published`` and bind it to ``content_version_id``.
   689Ôćĺ
   690Ôćĺ        Pre-state: ``in_review`` with every required :class:`ReviewStage`
   691Ôćĺ        approved (see :attr:`is_review_complete`).
   692Ôćĺ        Post-state: ``published`` with ``content_version_id`` and
   693Ôćĺ        ``published_at`` set.
   694Ôćĺ
   695Ôćĺ        The application service is responsible, in the same
   696Ôćĺ        transaction, for marking the bound :class:`ContentVersion` as
   697Ôćĺ        published (see :meth:`ContentVersion.publish`) ÔÇö that step
   698Ôćĺ        lives outside this aggregate because ContentVersion has its
   699Ôćĺ        own repository and is not part of the ContentPack aggregate
   700Ôćĺ        boundary.
   701Ôćĺ
   702Ôćĺ        Args:
   703Ôćĺ            content_version_id: The :class:`ContentVersion` to bind
   704Ôćĺ                this pack's artifacts to. Must already exist (the
   705Ôćĺ                application service creates it or selects an existing
   706Ôćĺ                one).
   707Ôćĺ            now: Optional timestamp (for testing); defaults to UTC now.
   708Ôćĺ
   709Ôćĺ        Raises:
   710Ôćĺ            ContentPackAlreadyPublished: If the pack is already
   711Ôćĺ                ``published``.
   712Ôćĺ            ContentPackAlreadyRejected: If the pack is ``rejected``.
   713Ôćĺ            ContentPackNotInReview: If the pack is ``draft`` (must be
   714Ôćĺ                submitted for review first).
   715Ôćĺ            ContentPackReviewIncomplete: If not every required review
   716Ôćĺ                stage has approved.
   717Ôćĺ        """
   718Ôćĺ        self._assert_not_terminal()
   719Ôćĺ        if self._status != ContentStatus.IN_REVIEW:
   720Ôćĺ            raise ContentPackNotInReview(self._id, self._status.value)
   721Ôćĺ        if not self.is_review_complete:
   722Ôćĺ            raise ContentPackReviewIncomplete(self._id, self.missing_stages)
   723Ôćĺ
   724Ôćĺ        timestamp = now or _utcnow()
   725Ôćĺ        self._status = ContentStatus.PUBLISHED
   726Ôćĺ        self._content_version_id = content_version_id
   727Ôćĺ        self._published_at = timestamp
   728Ôćĺ        self._record_event(
   729Ôćĺ            ContentPackPublished(
   730Ôćĺ                content_pack_id=self._id,
   731Ôćĺ                subject_id=self._subject_id,
   732Ôćĺ                content_version_id=content_version_id,
   733Ôćĺ                published_at=timestamp,
   734Ôćĺ            )
   735Ôćĺ        )
   736Ôćĺ
